Note:

A greedy approach can also be used to solve this problem. By sorting both the questions array and the applicants array, we can then iterate through each applicant using a for loop and assign them the question rating in ascending order. Although its not efficient compared to the solution provided. Check out the explanation here.


            #include <bits/stdc++.h>
            using namespace std;
            
            void solve() {
                int n;
                cin >> n;
                vector<pair<int, int>> applicants(n);
                vector<int> questions(n);
            
                // Reading applicant ratings and storing with original index
                for (int i = 0; i < n; ++i) {
                    cin >> applicants[i].first;
                    applicants[i].second = i;
                }
            
                // Reading question ratings in jumbled order
                for (int i = 0; i < n; ++i) {
                    cin >> questions[i];
                }
            
                // Sorting both applicants and questions based on ratings
                sort(applicants.begin(), applicants.end());
                sort(questions.begin(), questions.end());
            
                // Array to store the matched question index for each applicant
                vector<int> matches(n);
            
                // Matching questions to applicants and storing question ratings directly
               for (int i = 0; i < n; ++i) {
                matches[applicants[i].second] = questions[i];   
               }
            
               // Outputting the correct order of question ratings
               for (int i = 0; i < n; ++i) {
                cout << matches[i] << " ";
               }
                cout << "\n";
            }
            
            int main() {
                ios_base::sync_with_stdio(false); cin.tie(NULL);
                int t;
                cin >> t;
                while (t--) {
                    solve();
                }
                return 0;
            }
